## 筛选
```dataviewjs
const container = dv.container;//设置一个选项容器变量
const selectElement = dv.el('select');//设置选项变量
let categorySet = new Set(dv.pages('"策划知识库/Books"').category); //找到路径页面下的category标签

for (let category of categorySet) {
    const option_category = dv.el('option', `${category}`);
    selectElement.appendChild(option_category);
}//筛选对应category标签并分组

container.appendChild(selectElement);//选项容器添加选项

const dynamicContentContainer = dv.el('div');
container.appendChild(dynamicContentContainer);//设置筛选容器

selectElement.addEventListener('change', function() {
    %% dynamicContentContainer.innerHTML = '';

    let selectedValue = this.value;
    let category_pages = dv.pages(`${selectedValue}`);

    dynamicContentContainer.appendChild(dv.el('span', `<br><b>${selectedValue}</b><br>`));

    for (let page of category_pages) {
    dynamicContentContainer.appendChild(dv.el('span', `${page.file.link}<br>`));//列出满足选项 %%
    }
});

```
